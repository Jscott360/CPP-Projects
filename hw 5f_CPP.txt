#include<iostream>
using namespace std;
bool isVowel(char ch) 
{
    ch=toupper(ch);   
    return(ch=='A' || ch=='E' || ch=='I' || ch=='O' || ch=='U' );
}
int  countVowel(char str[])
{
   int i=0,count=0;  
   while(str[i]!='\0')
   {
       if(isVowel(str[i])) 
          {
              ++count;  
          }
       i++; 
   }
   return count;  
}
int main()
{
    char statement[20]=""; 
    cout<<"Enter a statement:\n\n";
    cin.getline(statement,20); 
    cout<<endl<<"There are "<<countVowel(statement)<<" vowel in the statement."<<endl;
    return 0;
}
